export const categories = pgTable('categories', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: varchar('name', { length: 100 }).notNull(),
  slug: varchar('slug', { length: 100 }).notNull().unique(),
  description: text('description'),
  displayOrder: integer('display_order').notNull().default(0),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
})

export const topics = pgTable('topics', {
  id: uuid('id').primaryKey().defaultRandom(),
  categoryId: uuid('category_id').notNull().references(() => categories.id, { onDelete: 'cascade' }),
  name: varchar('name', { length: 100 }).notNull(),
  slug: varchar('slug', { length: 100 }).notNull(),
  description: text('description'),
  displayOrder: integer('display_order').notNull().default(0),
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
}, (table) => [unique('topics_category_slug_unique').on(table.categoryId, table.slug)])

export const tags = pgTable('tags', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: varchar('name', { length: 100 }).notNull().unique(),
  slug: varchar('slug', { length: 100 }).notNull().unique(),
  displayName: varchar('display_name', { length: 100 }).notNull(),
})

  <name>Task 1: Taxonomy server actions, sortable list component, and tests</name>
  <files>src/lib/actions/taxonomy.ts, src/components/admin/sortable-list.tsx, src/__tests__/admin-taxonomy.test.ts</files>
  <action>
1. Create `src/lib/actions/taxonomy.ts`:
   - `'use server'` directive.
   - **Categories:** `createCategory(formData)` -- insert with name, slug (auto-gen from name), description, displayOrder (max+1). `updateCategory(id, formData)` -- update fields. `deleteCategory(id)` -- delete (cascade removes topics). `reorderCategories(orderedIds: string[])` -- accept array of category IDs in new order, update displayOrder for each (index = order).
   - **Topics:** `createTopic(formData)` -- insert with name, slug, description, categoryId, displayOrder. `updateTopic(id, formData)` -- update fields. `deleteTopic(id)` -- delete (cascade removes channelTopics). `reorderTopics(categoryId: string, orderedIds: string[])` -- reorder topics within a category.
   - **Tags:** `createTag(formData)` -- insert with name, slug, displayName. `updateTag(id, formData)` -- update. `deleteTag(id)` -- delete (cascade removes channelTags).
   - All mutations call `revalidatePath('/admin/categories')`, `revalidatePath('/admin/topics')`, `revalidatePath('/admin/tags')`, and `revalidatePath('/')` to refresh public pages.
   - Redirect back to list page after create/update/delete.

2. Create `src/components/admin/sortable-list.tsx`:
   - Client component (`'use client'`).
   - Generic reusable component using `@dnd-kit/core` + `@dnd-kit/sortable`.
   - Props: `items: { id: string; [key: string]: any }[]`, `onReorder: (orderedIds: string[]) => void`, `renderItem: (item) => ReactNode`.
   - Uses `DndContext` with `closestCenter` collision detection, `PointerSensor` and `KeyboardSensor`.
   - `SortableContext` with `verticalListSortingStrategy`.
   - Each item wrapped in a `SortableItem` using `useSortable` hook.
   - On `DragEnd`, compute new order with `arrayMove` and call `onReorder` with the new ID array.
   - Include a drag handle icon (GripVertical from lucide-react).
   - Smooth transition animations via `CSS.Transform.toString(transform)`.

3. Create `src/__tests__/admin-taxonomy.test.ts`:
   - Test slug generation from category/topic/tag names.
   - Test reorder logic: given [A, B, C] and moving A to position 2, result is [B, A, C].
   - Test that displayOrder values are sequential (0, 1, 2...) after reorder.
  </action>
